 
 Router0 Configuration
 
enable
configure terminal
 Configure router interfaces with IPs
interface GigabitEthernet0/0
 ip address 192.168.10.1 255.255.255.0
 no shutdown
exit

interface GigabitEthernet0/1
 ip address 192.168.20.1 255.255.255.0
 no shutdown
exit

interface GigabitEthernet0/2
 ip address 192.168.30.1 255.255.255.0
 no shutdown
exit
 ACL Configuration

ip access-list extended ADMIN_ICMP
  deny icmp 192.168.20.0 0.0.0.255 192.168.10.0 0.0.0.255 echo
  deny icmp 192.168.30.0 0.0.0.255 192.168.10.0 0.0.0.255 echo
  permit icmp any any echo-reply
  permit ip any any
exit

 Apply ACL inbound on Library and Lab interfaces
interface GigabitEthernet0/1
 ip access-group ADMIN_ICMP in
exit

interface GigabitEthernet0/2
 ip access-group ADMIN_ICMP in
exit

end
write memory
